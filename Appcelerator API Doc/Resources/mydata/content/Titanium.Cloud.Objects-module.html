Titanium.Cloud.Objects

module

Summary

Provides methods for accessing ACS custom objects.


Code Examples

Create Custom Object

This example creates a new custom object and checks the response.

Cloud.Objects.create({
    classname: 'cars',
    fields: {
        make: 'nissan',
        color: 'blue',
        year: 2005
    }
}, function (e) {
    if (e.success) {
        var car = e.cars[0];
        alert('Success:\\\
' +
            'id: ' + car.id + '\\\
' +
            'make: ' + car.make + '\\\
' +
            'color: ' + car.color + '\\\
' +
            'year: ' + car.year + '\\\
' +
            'created_at: ' + car.created_at);
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});
