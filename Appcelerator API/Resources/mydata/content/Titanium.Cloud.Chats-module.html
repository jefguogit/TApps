Titanium.Cloud.Chats

module

Summary

Provides methods for accessing ACS chat messages.


Code Examples

Create Chat Message

This example sends a chat message to a list of users and checks the response.

Cloud.Chats.create({
    to_ids: ids.join(','),
    message: 'Good morning'
}, function (e) {
    if (e.success) {
        for (var i = 0; i < e.chats.length; i++) {
            var chat = e.chats[i];
            alert('Success:\\\
' +
                'From: ' + chat.from.first_name + ' ' + chat.from.last_name + '\\\
' +
                'Updated: ' + chat.updated_at + '\\\
' +
                'Message: ' + chat.message);
        }
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});

Get Chat Groups

This example requests a lists of chat groups that the current user belongs to and checks the response.

Cloud.Chats.getChatGroups(function (e) {
    if (e.success) {
        for (var i = 0; i < e.chats.length; i++) {
            var group = e.chat_groups[i];
            alert('Success:\\\
' +
                'id: ' + group.id + '\\\
' +
                'created_at: ' + group.created_at + '\\\
' +
                'number of users: ' + group.participate_users.length);
        }
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});
