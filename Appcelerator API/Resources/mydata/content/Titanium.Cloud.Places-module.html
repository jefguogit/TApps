Titanium.Cloud.Places

module

Summary

Provides methods for accessing ACS places.


Code Examples

Create Place

This example creates a new place and checks the response.

Cloud.Places.create({
    name: 'Appcelerator HQ',
    state: 'California',
    website: 'http://www.appcelerator.com'
}, function (e) {
    if (e.success) {
        var place = e.places[0];
        alert('Success:\\\
' +
            'id: ' + place.id + '\\\
' +
            'name: ' + place.name + '\\\
' +
            'updated_at: ' + place.updated_at);
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});

Search for Places

This example searches for places and checks the response.

Cloud.Places.search({
    latitude: lat,
    longitude: lon
}, function (e) {
    if (e.success) {
        alert('Success:\\\
' +
            'Count: ' + e.places.length);
        for (var i = 0; i < e.places.length; i++) {
            var place = e.places[i];
            alert('id: ' + place.id + '\\\
' +
                  'name: ' + place.name + '\\\
' +
                  'longitude: ' + place.longitude + '\\\
' +
                  'latitude: ' + place.latitude + '\\\
' +
                  'updated_at: ' + place.updated_at);
        }
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});

Query for Places

    page: 1,
    per_page: 20,
