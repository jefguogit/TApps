Titanium.Cloud.Reviews

module

Summary

Provides methods for accessing ACS reviews.


Code Examples

Create Review

This example creates a new review and checks the response.

Cloud.Reviews.create({
    post_id: savedPostId,
    rating: 1,
    content: 'Good'
}, function (e) {
    if (e.success) {
        var review = e.reviews[0];
        alert('Success:\\\
' +
            'id: ' + review.id + '\\\
' +
            'rating: ' + review.rating + '\\\
' +
            'content: ' + review.content + '\\\
' +
            'updated_at: ' + review.updated_at);
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});

Query for Reviews

This example requests a list of reviews and checks the response.

Cloud.Reviews.query({
    page: 1,
    per_page: 20,
    where: {
        rating: { '$gt': 5.0 }
    }
}, function (e) {
    if (e.success) {
        alert('Success:\\\
' +
            'Count: ' + e.reviews.length);
        for (var i = 0; i < e.reviews.length; i++) {
            var review = e.reviews[i];
            alert('id: ' + review.id + '\\\
' +
                'id: ' + review.id + '\\\
' +
                'rating: ' + review.rating + '\\\
' +
                'content: ' + review.content + '\\\
' +
                'updated_at: ' + review.updated_at);
        }
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});
