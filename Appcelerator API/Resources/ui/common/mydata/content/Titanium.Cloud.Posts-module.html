Titanium.Cloud.Posts

module

Summary

Provides methods for accessing ACS posts.


Code Examples

Create Post

This example creates a new post and checks the response.

Cloud.Posts.create({
    content: 'Man Walks On Moon',
    title: 'News of the day',
    photo: Titanium.Filesystem.getFile('photo.jpg')
}, function (e) {
    if (e.success) {
        var post = e.posts[0];
        alert('Success:\\\
' +
            'id: ' + post.id + '\\\
' +
            'title: ' + post.title + '\\\
' +
            'content: ' + post.content + '\\\
' +
            'updated_at: ' + post.updated_at);
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});

Query for Posts

This example requests a list of posts and checks the response.

Cloud.Posts.query({
    page: 1,
    per_page: 20,
    where: {
        reviews_count: { '$gt': 1.0 }
    }
}, function (e) {
    if (e.success) {
        alert('Success:\\\
' +
            'Count: ' + e.posts.length);
        for (var i = 0; i < e.posts.length; i++) {
            var post = e.posts[i];
            alert('id: ' + post.id + '\\\
' +
                'id: ' + post.id + '\\\
' +
                'title: ' + post.title + '\\\
' +
                'content: ' + post.content + '\\\
' +
                'updated_at: ' + post.updated_at);
        }
    } else {
        alert('Error:\\\
' +
            ((e.error && e.message) || JSON.stringify(e)));
    }
});
